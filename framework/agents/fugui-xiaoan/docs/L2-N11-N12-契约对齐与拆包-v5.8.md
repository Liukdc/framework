## 工具 contract-in/out Schema（N11产出）

### 必用工具完整定义

```json
{
  "tools": {
    "insert_record": {
      "contractIn": {
        "category": "string（种类，用户原话）",
        "amount": "number（金额，正则匹配后解析）",
        "date": "string（日期 YYYY-MM-DD，模型做时间语义解析）",
        "quantity": "number（数量，默认1）",
        "unit": "string|null（单位，默认null）"
      },
      "contractOut": {
        "success": true,
        "recordId": "number（新增记录ID）"
      },
      "sqlTemplate": "INSERT INTO records (category, amount, date, quantity, unit) VALUES (?, ?, ?, ?, ?)"
    },
    "query_records": {
      "contractIn": {
        "startDate": "string|null（开始日期 YYYY-MM-DD）",
        "endDate": "string|null（结束日期 YYYY-MM-DD）",
        "category": "string|null（种类过滤）",
        "minAmount": "number|null（最低金额）",
        "maxAmount": "number|null（最高金额）"
      },
      "contractOut": {
        "records": [{ "id": "number", "category": "string", "amount": "number", "date": "string", "quantity": "number", "unit": "string|null" }],
        "total": "number（记录总数）"
      },
      "sqlTemplate": "SELECT * FROM records WHERE date BETWEEN ? AND ? [AND category=?] [AND amount>=?] [AND amount<=?]"
    },
    "delete_record": {
      "contractIn": {
        "recordId": "number（要删除的记录ID）"
      },
      "contractOut": {
        "success": true,
        "deletedRecord": { "category": "string", "amount": "number", "date": "string" }
      },
      "sqlTemplate": "DELETE FROM records WHERE id = ?"
    },
    "query_total": {
      "contractIn": {
        "timeRange": "string（时段描述，如'this_month'/'last_month'/'2026-07'——DET内部时间解析）"
      },
      "contractOut": {
        "total": "number（支出总额）"
      },
      "sqlTemplate": "SELECT SUM(amount) as total FROM records WHERE date BETWEEN ? AND ?"
    },
    "query_unit_price": {
      "contractIn": {
        "item": "string（物品名，如'猪肉'）",
        "date": "string（日期描述，如'yesterday'——DET内部时间解析）"
      },
      "contractOut": {
        "unitPrice": "number（单价 = amount ÷ quantity）",
        "unit": "string（如'斤'）",
        "amount": "number",
        "quantity": "number"
      },
      "sqlTemplate": "SELECT amount, quantity, unit FROM records WHERE category=? AND date=?"
    }
  },
  "metaTool": {
    "search_tools": {
      "contractIn": { "query": "string（能力描述）" },
      "contractOut": { "tools": [{ "name": "string", "description": "string" }] }
    }
  }
}
```
